JSE_TOKENDST (on)

JSE_OPERATOR_OVERLOADING (on)

JSE_ENABLE_DYNAMETH (off)

JSE_GETFILENAMELIST (off)

JSE_BREAKPOINT_TEST (off)

JSE_TASK_SCHEDULER (on)

JSE_INCLUDE (on)

JSE_DEFINE (on)

JSE_CONDITIONAL_COMPILE (on)

JSE_SECUREJSE (on)

JSE_MAIN_ARGC_ARGV (on)

JSE_TOSOURCE (on)

JSE_NAMED_PARAMS (on)

JSE_MILLENIUM (off)
